import pygame import random # Initialize pygame pygame.init() # Set up display WIDTH, HEIGHT = 800, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption('The Fading Relationship') # Colors WHITE = (255, 255, 255) DARK_RED = (139, 0, 0) LIGHT_GRAY = (211, 211, 211) BLACK = (0, 0, 0) FADING_COLOR = (150, 0, 0) # Fonts font = pygame.font.SysFont('Arial', 30) # Game variables player_width = 60 player_height = 60 player_x = WIDTH // 2 - player_width // 2 player_y = HEIGHT - player_height - 10 player_speed = 5 falling_objects = [] object_speed = 5 game_over = False score = 0 time_left = 100 # The time counter representing the final chance # Set up clock for frame rate clock = pygame.time.Clock() # Function to draw the player def draw_player(x, y): pygame.draw.rect(screen, DARK_RED, (x, y, player_width, player_height)) # Function to draw falling objects def draw_falling_objects(objects): for obj in objects: pygame.draw.circle(screen, FADING_COLOR, obj, 20) # Function to generate new objects def generate_new_object(): x = random.randint(50, WIDTH - 50) return [x, 0] # Function to draw the game over screen def draw_game_over(): game_over_text = font.render('The relationship is lost...', True, LIGHT_GRAY) screen.blit(game_over_text, (WIDTH // 2 - 150, HEIGHT // 2)) final_message = font.render('No more chances. Itâ€™s over.', True, LIGHT_GRAY) screen.blit(final_message, (WIDTH // 2 - 170, HEIGHT // 2 + 40)) # Function to draw the background fading effect def draw_fading_background(): gradient = pygame.Surface((WIDTH, HEIGHT)) for i in range(HEIGHT): color = [max(255 - i // 2, 0) for _ in range(3)] pygame.draw.line(gradient, tuple(color), (0, i), (WIDTH, i)) screen.blit(gradient, (0, 0)) # Main game loop while True: screen.fill(WHITE) # Draw fading background draw_fading_background() # Handle events for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() quit() # Check for key presses keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] and player_x > 0: player_x -= player_speed if keys[pygame.K_RIGHT] and player_x < WIDTH - player_width: player_x += player_speed # Add new falling objects periodically if random.randint(1, 100) <= 5: falling_objects.append(generate_new_object()) # Move falling objects down for obj in falling_objects: obj[1] += object_speed # Check for collisions with the player for obj in falling_objects[:]: if (obj[0] - player_x) ** 2 < (player_width // 2) ** 2 and (obj[1] - player_y) ** 2 < (player_height // 2) ** 2: falling_objects.remove(obj) score += 1 # Objects that fall out of screen if obj[1] > HEIGHT: falling_objects.remove(obj) # Decrease the remaining time (symbolizing time running out in the relationship) time_left -= 0.05 # Draw everything draw_player(player_x, player_y) draw_falling_objects(falling_objects) # Show score and time left score_text = font.render(f'Score: {score}', True, BLACK) screen.blit(score_text, (10, 10)) time_text = font.render(f'Time Left: {int(time_left)}', True, BLACK) screen.blit(time_text, (WIDTH - 150, 10)) # If the player has failed to catch enough objects or time runs out, end the game if time_left <= 0: game_over = True if game_over: draw_game_over() pygame.display.update() pygame.time.wait(2000) # Wait for a short time to show the game over screen pygame.quit() quit() pygame.display.update() clock.tick(60)
